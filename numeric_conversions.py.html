def hex_char_decode(digit):
    value = 0
    if digit == '0':
        value = 0
    if digit == '1':
        value = 1
    if digit == '2':
        value = 2
    if digit == '3':
        value = 3
    if digit == '4':
        value = 4
    if digit == '5':
        value = 5
    if digit == '6':
        value = 6
    if digit == '7':
        value = 7
    if digit == '8':
        value = 8
    if digit == '9':
        value = 9
# below gives number value to a letter in hexadecimal system
    if digit == 'A' or "a":
        value = 10
    if digit == 'B' or 'b':
        value = 11
    if digit == 'C' or 'c':
        value = 12
    if digit == 'D' or 'd':
        value = 13
    if digit == 'E' or 'e':
        value = 14
    if digit == 'F' or 'f':
        value = 15

    return value


def hex_string_decode(hex):
    if hex[0] == '0' and hex[1] == 'x':
        hex = hex[2:]
    bit_index = len(hex)
    total = 0
    for digit in hex:
        total += hex_string_decode(digit) * 16 ** bit_index
        bit_index -= 1
    return total


def binary_string_decode(binary):
    if binary[0] == '0' or binary[2] == 'b':
        binary = binary[2:]
    bit_index = len(binary)
    total = 0
    for digit in binary:
        total += binary_string_decode(digit)


def binary_to_hex(binary):
    res = ""
    while binary > 0:
        remainder = binary % 16
        binary //= 16
        res = str(remainder) + res

    return res
